#include <cstdio>
#include <cmath>

const int MAXN = 100000;

int n;
double p[MAXN + 1], f[MAXN + 1];
double ex1[MAXN + 1], ex2[MAXN + 1];

inline void dp() {
    f[0] = 0;
    for (int i = 1; i <= n; i++) {
        ex1[i] = (ex1[i - 1] + 1) * p[i];
        ex2[i] = (ex2[i - 1] + 2 * ex1[i - 1] + 1) * p[i];
        double t = (3 * ex2[i - 1] + 3 * ex1[i - 1] + 1) * p[i];
        f[i] = f[i - 1] + t;
    }
}

int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%lf", &p[i]);

    dp();

    printf("%.1lf\n", floor(f[n] * 10 + 0.5) / 10);

    return 0;
}
