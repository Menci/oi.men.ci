#include <cstdio>
#include <cstring>
#include <algorithm>

const int MAXN = 2000 + 10;

int C[MAXN + 1][MAXN + 1], cnt[MAXN + 1][MAXN + 1];

int main() {
    freopen("problem.in", "r", stdin);
    freopen("problem.out", "w", stdout);

    int t, k;
    scanf("%d %d", &t, &k);

    for (int i = 1; i <= MAXN; i++) {
        C[i][0] = C[i][i] = 1;
        for (int j = 1; j < i; j++) {
            C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % k;
        }

        for (int j = 1; j <= i; j++) {
            cnt[i][j] = cnt[i][j - 1];
            if (C[i][j] == 0) cnt[i][j]++;
        }
    }

    while (t--) {
        int n, m;
        scanf("%d %d", &n, &m);

        int ans = 0;
        for (int i = 1; i <= n; i++) {
            ans += cnt[i][std::min(i, m)];
        }

        printf("%d\n", ans);
    }

    fclose(stdin);
    fclose(stdout);

    return 0;
}
